BEGIN
    delete from product_parameters where mnemo = 'testDate';
    insert into product_parameters (id, config, copyto, mnemo, name, options, prty, required, value, lob, analytic, type, product_id, variant_id)
    with testdate as (
    select null config, 0 copyto, 'testDate' mnemo, 'Тестовая дата' name, 
   null options, 555 prty, 0 required, '30.01.2023' value, null as lob, null as analytic, 'DATE' as type from dual)
   select PRODUCT_PARAMETERS_ID_SEQ.nextval id, selrow.* from (
   select testdate.*, p.id product_id, null variant_id from testdate, products p
    ) selrow;
   COMMIT;
END;
